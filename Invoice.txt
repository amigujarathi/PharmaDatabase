CREATE TABLE INVOICES
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    PURCHASE_ORDER_ID RAW(16) NOT NULL,
    SUPPLIER_ID RAW(16) NOT NULL,
    SENDER_ORGANIZATION_ID RAW(16) NOT NULL,
    RECEIVER_ORGANIZATION_ID RAW(16) NOT NULL,
    RATE NUMBER(18,2) NOT NULL,
    QUANTITY NUMBER(18,6) NOT NULL,
    FRIEGHT_AMOUNT NUMBER(18,2) NOT NULL,
    PAYMENT_TERMS VARCHAR2(128),
    INVOICE_DATE TIMESTAMP(6) NOT NULL,
    ISSUE_DATE TIMESTAMP(6) NOT NULL,
    PAYMENT_DUE_DATE TIMESTAMP(6) NOT NULL,
    CURRENCY_ID RAW(16) NOT NULL,
    TAX VARCHAR2(32),
    TAX_RATE NUMBER(18,2),
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    UNIT_OF_MEASUREMENT_ID RAW(16) NOT NULL,
    INCOTERM VARCHAR2(256) NOT NULL,
    CONSTRAINT INVOICES_PO_FK FOREIGN KEY (PURCHASE_ORDER_ID) REFERENCES PURCHASE_ORDERS (ID),
    CONSTRAINT INVOICES_SUPPLIERS_FK FOREIGN KEY (SUPPLIER_ID) REFERENCES SUPPLIERS (ID),
    CONSTRAINT INVOICES_ORGANIZATIONS_FK_1 FOREIGN KEY (SENDER_ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID),
    CONSTRAINT INVOICES_ORGANIZATIONS_FK_2 FOREIGN KEY (RECEIVER_ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID),
    CONSTRAINT INVOICES_CURRENCY_FK FOREIGN KEY (CURRENCY_ID) REFERENCES COUNTRIES (ID),
    CONSTRAINT INVOICES_UOM_FK FOREIGN KEY (UNIT_OF_MEASUREMENT_ID) REFERENCES UNIT_OF_MEASUREMENTS (ID)
);
CREATE UNIQUE INDEX INVOICES_UK1 ON INVOICES (PURCHASE_ORDER_ID);
