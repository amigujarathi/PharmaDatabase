CREATE TABLE QUOTATIONS
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    ENQUIRY_ID RAW(16) NOT NULL,
    SENDER_ORGANIZATION_ID RAW(16) NOT NULL,
    RECEIVER_ORGANIZATION_ID RAW(16) NOT NULL,
    RATE NUMBER(18,2) NOT NULL,
    QUANTITY NUMBER(18,6) NOT NULL,
    PAYMENT_TERMS VARCHAR2(64),
    LEAD_TIME VARCHAR2(128),
    TAX VARCHAR2(64) NOT NULL,
    TAX_RATE NUMBER(18,2) NOT NULL,
    OFFER_VALIDITY TIMESTAMP(6),
    MESSAGE VARCHAR2(128),
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    INCOTERM VARCHAR2(256),
    STATUS VARCHAR2(64) DEFAULT 'PENDING' NOT NULL,
    FRIEGHT_TYPE VARCHAR2(64) DEFAULT 'PAID' NOT NULL,
    PACKAGING VARCHAR2(64) DEFAULT 'STANDARD' NOT NULL,
    PACKAGING_SPECIAL VARCHAR2(256),
    PAYMENT_TERMS_OTHER VARCHAR2(256),
    SHIPMENT_BY VARCHAR2(64) DEFAULT 'AIR',
    CONSTRAINT QUOTATIONS_ENQUIRIES_FK FOREIGN KEY (ENQUIRY_ID) REFERENCES ENQUIRIES (ID),
    CONSTRAINT QUOTATIONS_ORGANIZATIONS_FK_1 FOREIGN KEY (SENDER_ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID),
    CONSTRAINT QUOTATIONS_ORGANIZATIONS_FK_2 FOREIGN KEY (RECEIVER_ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID)
);
CREATE UNIQUE INDEX QUOTATIONS_UK1 ON QUOTATIONS (ENQUIRY_ID);
