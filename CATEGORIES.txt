CREATE TABLE CATEGORIES
(
  ID RAW(16) PRIMARY KEY NOT NULL,
  CATEGORY VARCHAR2(256) NOT NULL,
  DISCRIPTION VARCHAR2(512),
  DELETED NUMBER(1) DEFAULT 0 NOT NULL,
  CREATED TIMESTAMP(6),
  UPDATED TIMESTAMP(6),
  CREATED_BY VARCHAR2(256),
  UPDATED_BY VARCHAR2(256)
);
CREATE UNIQUE INDEX CATEGORIES_UK ON CATEGORIES (CATEGORY);
