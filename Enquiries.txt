CREATE TABLE ENQUIRIES
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    SUPPLIER_ID RAW(16) NOT NULL,
    QUANTITY NUMBER(18,6) NOT NULL,
    LOCATION_ID RAW(16) NOT NULL,
    CURRENCY_ID RAW(16) NOT NULL,
    SHIPMENT_BY VARCHAR2(64) DEFAULT 'AIR' NOT NULL,
    MESSAGE VARCHAR2(128),
    SENDER_ORGANIZATION_ID RAW(16) NOT NULL,
    RECEIVER_ORGANIZATION_ID RAW(16) NOT NULL,
    STATUS VARCHAR2(32) DEFAULT 'SENT' NOT NULL,
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    UNIT_OF_MEASUREMENT_ID RAW(16) NOT NULL,
    INCOTERM VARCHAR2(256),
    ENQUIRY_LOCATION VARCHAR2(1024),
    CONSTRAINT ENQUIRIES_SUPPLIERS_FK FOREIGN KEY (SUPPLIER_ID) REFERENCES SUPPLIERS (ID),
    CONSTRAINT ENQUIRIES_LOCATIONS_FK FOREIGN KEY (LOCATION_ID) REFERENCES LOCATIONS (ID),
    CONSTRAINT ENQUIRIES_CURRENCIES_FK FOREIGN KEY (CURRENCY_ID) REFERENCES COUNTRIES (ID),
    CONSTRAINT ENQUIRIES_ORG_FK_1 FOREIGN KEY (SENDER_ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID),
    CONSTRAINT ENQUIRIES_ORG_FK_2 FOREIGN KEY (RECEIVER_ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID),
    CONSTRAINT ENQUIRIES_UOM_FK FOREIGN KEY (UNIT_OF_MEASUREMENT_ID) REFERENCES UNIT_OF_MEASUREMENTS (ID)
);
