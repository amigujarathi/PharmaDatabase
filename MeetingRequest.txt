CREATE TABLE MEETINGREQUESTS
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    USER_ID RAW(16) NOT NULL,
    COMPANY VARCHAR2(256),
    EMAIL_TO VARCHAR2(256),
    SUBJECT VARCHAR2(256),
    BODY VARCHAR2(1024),
    SENDER VARCHAR2(256),
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    FIRST_NAME VARCHAR2(256),
    LAST_NAME VARCHAR2(256),
    CONSTRAINT MEETING_REQUESTS_FK FOREIGN KEY (USER_ID) REFERENCES USERS (ID)
);
