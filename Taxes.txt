CREATE TABLE TAXES
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    TAX VARCHAR2(128) NOT NULL,
    PERCENT NUMBER(18,2) NOT NULL,
    DESCRIPTION VARCHAR2(256),
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    ORGANIZATION_ID RAW(16),
    CONSTRAINT TAXES_ORGANIZATIONS_FK FOREIGN KEY (ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID)
);
