CREATE TABLE BUYINGREQUIREMENTS
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    PRODUCT_NAME VARCHAR2(3072),
    NAME VARCHAR2(256),
    MOBILE VARCHAR2(64),
    EMAIL VARCHAR2(256),
    CITY VARCHAR2(128),
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256)
);
