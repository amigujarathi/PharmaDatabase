CREATE TABLE USER_FORGET_PASSWORDS
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    USER_ID RAW(16),
    USERNAME VARCHAR2(256),
    EMAIL VARCHAR2(256),
    MOBILE VARCHAR2(256),
    PASSWORD VARCHAR2(256),
    HASHCODE VARCHAR2(256),
    FLAG NUMBER(1) DEFAULT 0 NOT NULL,
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    CONSTRAINT USER_FORGET_PASSWORDS_USERS_FK FOREIGN KEY (USER_ID) REFERENCES USERS (ID)
);
