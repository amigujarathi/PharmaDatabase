CREATE TABLE PURCHASE_ORDERS
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    QUOTATION_ID RAW(16) NOT NULL,
    SENDER_ORGANIZATION_ID RAW(16) NOT NULL,
    RECEIVER_ORGANIZATION_ID RAW(16) NOT NULL,
    RATE NUMBER(18,2) NOT NULL,
    QUANTITY NUMBER(18,6) NOT NULL,
    PAYMENT_TERMS VARCHAR2(64) NOT NULL,
    TAX VARCHAR2(32) NOT NULL,
    TAX_RATE NUMBER(18,2) NOT NULL,
    DELIVERY_SCHEDULE TIMESTAMP(6) NOT NULL,
    MESSAGE VARCHAR2(128),
    STATUS VARCHAR2(32) DEFAULT 'SENT' NOT NULL,
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    COMMERCIAL_INVOICE NUMBER(1) DEFAULT 0 NOT NULL,
    PACKAGING_LIST NUMBER(1) DEFAULT 0 NOT NULL,
    NON_PREF_CERT_ORIGIN NUMBER(1) DEFAULT 0 NOT NULL,
    BILL_OF_LADING NUMBER(1) DEFAULT 0 NOT NULL,
    HEALTH_CERTIFICATE NUMBER(1) DEFAULT 0 NOT NULL,
    PREF_CERT_ORIGIN NUMBER(1) DEFAULT 0 NOT NULL,
    MANUFACTURING_CERT NUMBER(1) DEFAULT 0 NOT NULL,
    CERT_ANALYSIS NUMBER(1) DEFAULT 0 NOT NULL,
    INSURANCE_POLICY NUMBER(1) DEFAULT 0 NOT NULL,
    AIRWAY_BILL NUMBER(1) DEFAULT 0 NOT NULL,
    INCOTERM VARCHAR2(256),
    FRIEGHT_TYPE VARCHAR2(64) DEFAULT 'PAID',
    DOC_OTHERS VARCHAR2(256),
    OTHERS NUMBER(1) DEFAULT 0 NOT NULL,
    SHIPMENT_BY VARCHAR2(64) DEFAULT 'AIR',
    CONSTRAINT PO_QUOTATIONS_FK FOREIGN KEY (QUOTATION_ID) REFERENCES QUOTATIONS (ID),
    CONSTRAINT PO_ORGANIZATIONS_FK_1 FOREIGN KEY (SENDER_ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID),
    CONSTRAINT PO_ORGANIZATIONS_FK_2 FOREIGN KEY (RECEIVER_ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID)
);
CREATE UNIQUE INDEX PURCHASE_ORDERS_UK1 ON PURCHASE_ORDERS (QUOTATION_ID);
