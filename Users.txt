CREATE TABLE USERS
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    FIRST_NAME VARCHAR2(256) NOT NULL,
    LAST_NAME VARCHAR2(256) NOT NULL,
    MIDDLE_NAME VARCHAR2(256),
    GENDER VARCHAR2(8) DEFAULT 'MALE' NOT NULL,
    MOBILE VARCHAR2(32) NOT NULL,
    EMAIL VARCHAR2(256) NOT NULL,
    USERNAME VARCHAR2(256) NOT NULL,
    PASSWORD VARCHAR2(256) NOT NULL,
    URL VARCHAR2(512),
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256)
);
CREATE UNIQUE INDEX USERS_MOBILE_UK ON USERS (MOBILE);
CREATE UNIQUE INDEX USERS_EMAIL_UK ON USERS (EMAIL);
CREATE UNIQUE INDEX USERS_USERNAME_UK ON USERS (USERNAME);
