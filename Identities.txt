CREATE TABLE IDENTITIES
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    IDENTITY VARCHAR2(128) NOT NULL,
    ORGANIZATION_ID RAW(16) NOT NULL,
    TYPE VARCHAR2(128) NOT NULL,
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    CONSTRAINT IDENTITIES_ORGANIZATIONS_FK FOREIGN KEY (ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID)
);
CREATE UNIQUE INDEX IDENTITIES_ORGANIZATION_UK ON IDENTITIES (ORGANIZATION_ID, TYPE);
