CREATE TABLE USER_MOBILES
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    USER_ID RAW(16) NOT NULL,
    MOBILE VARCHAR2(128),
    HASHCODE VARCHAR2(256),
    VERIFIED NUMBER(1) DEFAULT 0 NOT NULL,
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    CONSTRAINT USER_MOBILES_USERS_FK FOREIGN KEY (USER_ID) REFERENCES USERS (ID)
);
