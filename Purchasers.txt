CREATE TABLE PURCHASERS
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    ORGANIZATION_ID RAW(16) NOT NULL,
    PRODUCT_ID RAW(16) NOT NULL,
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    CONSTRAINT PURCHASERS_ORGANIZATIONS_FK FOREIGN KEY (ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID),
    CONSTRAINT PURCHASERS_PRODUCTS_FK FOREIGN KEY (PRODUCT_ID) REFERENCES PRODUCTS (ID)
);
CREATE UNIQUE INDEX PUR_PROD_ORG_UK ON PURCHASERS (ORGANIZATION_ID, PRODUCT_ID);
