CREATE TABLE LOCATIONS
(
    ID RAW(16) PRIMARY KEY NOT NULL,
    LOCATION VARCHAR2(128) NOT NULL,
    ADDRESS_LINE_1 VARCHAR2(256),
    ADDRESS_LINE_2 VARCHAR2(256),
    POSTAL_CODE VARCHAR2(32) NOT NULL,
    CITY VARCHAR2(64) NOT NULL,
    REGION VARCHAR2(64),
    COUNTRY_ID RAW(16) NOT NULL,
    ORGANIZATION_ID RAW(16) NOT NULL,
    DELETED NUMBER(1) DEFAULT 0 NOT NULL,
    CREATED TIMESTAMP(6),
    UPDATED TIMESTAMP(6),
    CREATED_BY VARCHAR2(256),
    UPDATED_BY VARCHAR2(256),
    STATUS NUMBER(1) DEFAULT 0 NOT NULL,
    CONSTRAINT LOCATIONS_COUNTRIES_FK FOREIGN KEY (COUNTRY_ID) REFERENCES COUNTRIES (ID),
    CONSTRAINT LOCATIONS_ORGANIZATIONS_FK FOREIGN KEY (ORGANIZATION_ID) REFERENCES ORGANIZATIONS (ID)
);
